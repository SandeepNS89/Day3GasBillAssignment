CREATE TABLE CUSTOMER(
 CUSTNO              INT CONSTRAINT CUST_PK PRIMARY KEY,
 CUSTNAME            VARCHAR(20) CONSTRAINT CUST_NOTNUL NOT NULL,
 GENDER		     CHAR(1) DEFAULT 'F',
 EMAIL		     VARCHAR(20) CONSTRAINT CUST_MAIL_UNIQ UNIQUE,
 PHONE		     INT CONSTRAINT CUST_PH_UNIQ UNIQUE,
 AADHAR		     INT CONSTRAINT CUST_AADHAR_UNIQ UNIQUE,
 DOB                 DATE CONSTRAINT CUST_CHK_DOB CHECK (DOB > '2004-03-31'),
 GROSSINCOME         INT CONSTRAINT,
 BRANCHNO            INT REFERENCES BANK(BRANCHNO)
 JOBNO              INT REFERENCES JOB(JOBNO)
 EMPNO              INT REFERENCES EMPLOYEE(EMPNO));


CREATE TABLE ACCOUNTDETAILS(
 BRANCHNO            INT CONSTRAINT BR_PK PRIMARY KEY,
 BRANCHNAME          VARCHAR(20) CONSTRAINT BR_NAME_NOTNUL NOT NULL,
 IFSC                VARCHAR(20) CONSTRAINT BR_IFSC_NOTNUL NOT NULL,
 BRANCHEMAIL         VARCHAR(20) CONSTRAINT BR_MAIL_UNIQ UNIQUE,
 PHONE		     INT CONSTRAINT BR_PH_UNIQ UNIQUE,
 JOBADDRESS          VARCHAR(50) CONSTRAINT BANK_ADD_NOTNUL NOT NULL, 
 ACCNO               INT CONSTRAINT BR_ACC_UNIQ UNIQUE;                 //AUTO GENERATE
 ACCTYPE             VARCHAR(10) CONSTRAINT BR_ACCTYPE_NOTNUL NOT NULL,
 DATEOFOPENING       DATE,
 BALANCE	     INT,
 CUSTNO              INT REFERENCES CUSTOMER(CUSTNO));


CREATE TABLE JOB(
 JOBNO               INT CONSTRAINT CUST_PK PRIMARY KEY,
 DESIGNATION         VARCHAR(20) CONSTRAINT JOB_DES_NOTNUL NOT NULL,
 DATEOFJOINING       DATE,
 EMPLOYER	     VARCHAR(20) CONSTRAINT JOB_EMP_NOTNUL NOT NULL,
 JOBADDRESS          VARCHAR(50) CONSTRAINT JOB_ADD_NOTNUL NOT NULL, 
 CUSTNO              INT REFERENCES CUSTOMER(CUSTNO));

CREATE TABLE ADDRESS(
 ADDNO               INT CONSTRAINT ADD_PK PRIMARY KEY,
 STREET              VARCHAR(20) CONSTRAINT ADD_STR_NOTNUL NOT NULL,
 CITY    	     VARCHAR(20) CONSTRAINT ADD_CITY_NOTNUL NOT NULL,
 COUNTRY             VARCHAR(20) CONSTRAINT ADD_CON_NOTNUL NOT NULL, 
 PINNO               INT CONSTRAINT ADD_PIN_UNIQ UNIQUE);

CREATE TABLE EMPLOYEE(
 EMPNO               INT CONSTRAINT EMP_PK PRIMARY KEY,
 EMPNAME             VARCHAR(20) CONSTRAINT EMP_NAME_NOTNUL NOT NULL,
 GENDER		     CHAR(1) DEFAULT 'F',
 EMPEMAIL            VARCHAR(20) CONSTRAINT EMP_MAIL_UNIQ UNIQUE,
 PHONE		     INT CONSTRAINT EMP_PH_UNIQ UNIQUE,
 DOJ                 DATE,
 BRANCHNO            INT REFERENCES BANK(BRANCHNO)
 CUSTNO              INT REFERENCES CUSTOMER(CUSTNO));
